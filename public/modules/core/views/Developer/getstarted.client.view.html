2. Raspberry Pi code as follows for POST method:

    import json
    import pycurl

    x = json.dumps({
            "meta":"PulseRate",
            "value":125,
            "utc":"1404701788", 
            "provider":"facebook",
            "user":"1234567890"
            })
    headers = ['Content-Type: application/json']
    url_dir = 'https://qh.azure-api.net/dev/Prototyping?subscription-key=<your subscription key>'
    c = pycurl.Curl()
    c.setopt(pycurl.URL, url_dir)
    c.setopt(pycurl.HTTPHEADER, headers)
    c.setopt(pycurl.POST, 1)
    c.setopt(pycurl.POSTFIELDS, '%s' % x)
    c.perform()
    print("Uploaded!")
